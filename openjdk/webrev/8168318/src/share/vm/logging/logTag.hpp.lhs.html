 133 #define PREFIX_LOG_TAG(T) (LogTag::_##T)
 134 
 135 // Expand a set of log tags to their prefixed names.
 136 // For error detection purposes, the macro passes one more tag than what is supported.
 137 // If too many tags are given, a static assert in the log class will fail.
 138 #define LOG_TAGS_EXPANDED(T0, T1, T2, T3, T4, T5, ...)  PREFIX_LOG_TAG(T0), PREFIX_LOG_TAG(T1), PREFIX_LOG_TAG(T2), \
 139                                                         PREFIX_LOG_TAG(T3), PREFIX_LOG_TAG(T4), PREFIX_LOG_TAG(T5)
 140 // The EXPAND_VARARGS macro is required for MSVC, or it will resolve the LOG_TAGS_EXPANDED macro incorrectly.
 141 #define EXPAND_VARARGS(x) x
 142 #define LOG_TAGS(...) EXPAND_VARARGS(LOG_TAGS_EXPANDED(__VA_ARGS__, _NO_TAG, _NO_TAG, _NO_TAG, _NO_TAG, _NO_TAG, _NO_TAG))
 143 
 144 // Log tags are used to classify log messages.
 145 // Each log message can be assigned between 1 to LogTag::MaxTags number of tags.
 146 // Specifying multiple tags for a log message means that only outputs configured
 147 // for those exact tags, or a subset of the tags with a wildcard, will see the logging.
 148 // Multiple tags should be comma separated, e.g. log_error(tag1, tag2)("msg").
 149 class LogTag : public AllStatic {
 150  public:
 151   // The maximum number of tags that a single log message can have.
 152   // E.g. there might be hundreds of different tags available,
 153   // but a specific log message can only be tagged with up to MaxTags of those.
 154   static const size_t MaxTags = 5;
 155 
 156   enum type {
 157     __NO_TAG,
 158 #define LOG_TAG(name) _##name,
 159     LOG_TAG_LIST
 160 #undef LOG_TAG
 161     Count
 162   };
 163 
 164   static const char* name(LogTag::type tag) {
 165     return _name[tag];
 166   }
 167 
 168   static LogTag::type from_string(const char *str);
 169 
 170  private:
 171   static const char* _name[];
 172 };
 173 
 174 typedef LogTag::type LogTagType;
