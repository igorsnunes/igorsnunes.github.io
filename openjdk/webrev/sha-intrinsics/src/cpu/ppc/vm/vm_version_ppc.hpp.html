  26 #ifndef CPU_PPC_VM_VM_VERSION_PPC_HPP
  27 #define CPU_PPC_VM_VM_VERSION_PPC_HPP
  28 
  29 #include "runtime/globals_extension.hpp"
  30 #include "runtime/vm_version.hpp"
  31 
  32 class VM_Version: public Abstract_VM_Version {
  33 protected:
  34   enum Feature_Flag {
  35     fsqrt,
  36     fsqrts,
  37     isel,
  38     lxarxeh,
  39     cmpb,
  40     popcntb,
  41     popcntw,
  42     fcfids,
  43     vand,
  44     lqarx,
  45     vcipher,
  46     vshasig,
  47     vpmsumb,
  48     tcheck,
  49     mfdscr,
  50     vsx,
  51     num_features // last entry to count features
  52   };
  53   enum Feature_Flag_Set {
  54     unknown_m             = 0,
  55     fsqrt_m               = (1 << fsqrt  ),
  56     fsqrts_m              = (1 << fsqrts ),
  57     isel_m                = (1 << isel   ),
  58     lxarxeh_m             = (1 << lxarxeh),
  59     cmpb_m                = (1 << cmpb   ),
  60     popcntb_m             = (1 << popcntb),
  61     popcntw_m             = (1 << popcntw),
  62     fcfids_m              = (1 << fcfids ),
  63     vand_m                = (1 << vand   ),
  64     lqarx_m               = (1 << lqarx  ),
  65     vcipher_m             = (1 << vcipher),
  66     vshasig_m             = (1 << vshasig),
  67     vpmsumb_m             = (1 << vpmsumb),
  68     tcheck_m              = (1 << tcheck ),
  69     mfdscr_m              = (1 << mfdscr ),
  70     vsx_m                 = (1 << vsx    ),
  71     all_features_m        = (unsigned long)-1
  72   };
  73 
  74   static bool _is_determine_features_test_running;
  75 
  76   static void print_features();
  77   static void determine_features(); // also measures cache line size
  78   static void config_dscr(); // Power 8: Configure Data Stream Control Register.
  79   static void determine_section_size();
  80   static void power6_micro_bench();
  81 public:
  82   // Initialization
  83   static void initialize();
  84 
  85   // Override Abstract_VM_Version implementation
  86   static bool use_biased_locking();
  87 
  88   static bool is_determine_features_test_running() { return _is_determine_features_test_running; }
  89   // CPU instruction support
  90   static bool has_fsqrt()   { return (_features & fsqrt_m) != 0; }
  91   static bool has_fsqrts()  { return (_features & fsqrts_m) != 0; }
  92   static bool has_isel()    { return (_features & isel_m) != 0; }
  93   static bool has_lxarxeh() { return (_features & lxarxeh_m) !=0; }
  94   static bool has_cmpb()    { return (_features & cmpb_m) != 0; }
  95   static bool has_popcntb() { return (_features & popcntb_m) != 0; }
  96   static bool has_popcntw() { return (_features & popcntw_m) != 0; }
  97   static bool has_fcfids()  { return (_features & fcfids_m) != 0; }
  98   static bool has_vand()    { return (_features & vand_m) != 0; }
  99   static bool has_lqarx()   { return (_features & lqarx_m) != 0; }
 100   static bool has_vcipher() { return (_features & vcipher_m) != 0; }
 101   static bool has_vshasig() { return (_features & vshasig_m) != 0; }
 102   static bool has_vpmsumb() { return (_features & vpmsumb_m) != 0; }
 103   static bool has_tcheck()  { return (_features & tcheck_m) != 0; }
 104   static bool has_mfdscr()  { return (_features & mfdscr_m) != 0; }
 105   static bool has_vsx()     { return (_features & vsx_m) != 0; }
 106 
 107   // Assembler testing
 108   static void allow_all();
 109   static void revert();
 110 
 111   // POWER 8: DSCR current value.
 112   static uint64_t _dscr_val;
 113 };
 114 
 115 #endif // CPU_PPC_VM_VM_VERSION_PPC_HPP
